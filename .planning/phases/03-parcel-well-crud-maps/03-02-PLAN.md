  - {% block map_scripts %} loads MapLibre GL JS CDN and map-engine.css
  - Defines MAP_CONFIG as inline <script> before map-engine.js:

  ```javascript
  var MAP_CONFIG = {
      title: 'Open Water Accounting Platform',
      center: [{{ center_lng }}, {{ center_lat }}],
      zoom: {{ zoom }},
      maxZoom: 18,
      fitToData: true,
      geojsonSources: [
          { id: 'parcels', url: '{% url "parcels:geojson" %}' },
          { id: 'wells', url: '{% url "wells:geojson" %}' },
          { id: 'pods', url: '{% url "surface:pods_geojson" %}' },
          { id: 'recharge', url: '{% url "recharge:sites_geojson" %}' }
      ],
      layers: [
          {
              id: 'parcels-fill', type: 'fill', source: 'parcels',
              label: 'Parcels',
              paint: { 'fill-color': '#1B7FAF', 'fill-opacity': 0.3 },
              swatch: 'fill', swatchColor: '#1B7FAF'
          },
          {
              id: 'parcels-outline', type: 'line', source: 'parcels',
              group: 'parcels', groupHidden: true,
              paint: { 'line-color': '#1B7FAF', 'line-width': 1.5 }
          },
          {
              id: 'wells-points', type: 'circle', source: 'wells',
              label: 'Wells',
              paint: { 'circle-radius': 5, 'circle-color': '#E4A317', 'circle-stroke-width': 1.5, 'circle-stroke-color': '#fff' },
              glow: { width: 12, opacity: 0.15 },
              swatch: 'dot', swatchColor: '#E4A317'
          },
          {
              id: 'pods-points', type: 'circle', source: 'pods',
              label: 'Points of Diversion',
              paint: { 'circle-radius': 5, 'circle-color': '#4ECDC4', 'circle-stroke-width': 1.5, 'circle-stroke-color': '#fff' },
              glow: { width: 12, opacity: 0.15 },
              swatch: 'dot', swatchColor: '#4ECDC4'
          },
          {
              id: 'recharge-points', type: 'circle', source: 'recharge',
              label: 'Recharge Sites',
              paint: { 'circle-radius': 6, 'circle-color': '#7B68EE', 'circle-stroke-width': 1.5, 'circle-stroke-color': '#fff' },
              glow: { width: 14, opacity: 0.15 },
              swatch: 'dot', swatchColor: '#7B68EE'
          }
      ],
      popups: {
          'parcels-fill': function(p) {
              return '<strong>' + (p.parcel_number || 'Unknown') + '</strong><br>' +
                     'Owner: ' + (p.owner_name || '—') + '<br>' +
                     'Area: ' + (p.area_acres ? p.area_acres + ' ac' : '—') + '<br>' +
                     'Status: ' + (p.status || '—') + '<br>' +
                     '<a href="/parcels/' + p.pk + '/" style="color:#E4A317">View details →</a>';
          },
          'wells-points': function(p) {
              return '<strong>' + (p.name || 'Unnamed Well') + '</strong><br>' +
                     'Reg ID: ' + (p.well_registration_id || '—') + '<br>' +
                     'Depth: ' + (p.depth_ft ? p.depth_ft + ' ft' : '—') + '<br>' +
                     'Status: ' + (p.status || '—') + '<br>' +
                     '<a href="/wells/' + p.pk + '/" style="color:#E4A317">View details →</a>';
          },
          'pods-points': function(p) {
              return '<strong>' + (p.name || 'Unknown POD') + '</strong><br>' +
                     'Stream: ' + (p.stream_name || '—') + '<br>' +
                     'Max rate: ' + (p.max_rate_cfs ? p.max_rate_cfs + ' cfs' : '—');
          },
          'recharge-points': function(p) {
              return '<strong>' + (p.name || 'Unknown Site') + '</strong><br>' +
                     'Type: ' + (p.site_type || '—') + '<br>' +
                     'Capacity: ' + (p.capacity_acre_feet ? p.capacity_acre_feet + ' AF' : '—');
          }
      },
      legend: [
          {
              title: 'Features',
              items: [
                  { label: 'Parcels', swatch: 'fill', color: '#1B7FAF' },
                  { label: 'Wells', swatch: 'dot', color: '#E4A317' },
                  { label: 'Points of Diversion', swatch: 'dot', color: '#4ECDC4' },
                  { label: 'Recharge Sites', swatch: 'dot', color: '#7B68EE' }
              ]
          }
      ]
  };
  ```
