# Open Water Accounting Platform - Scheduled Jobs
#
# Install:   make install-cron
# Verify:    make show-cron
#
# Set the two paths below to match YOUR deployment before installing:
#   OPENH2O_DIR      where you cloned the repo (the dir holding docker-compose.yml)
#   OPENH2O_LOG_DIR  a directory the cron user can write logs to
#
# Jobs:
#   run-sync.sh    - Resilient external-data sync (ensures container is up,
#                    logs to a writable path, alerts via ntfy on failure if
#                    OPENH2O_NTFY_URL is set).
#                    Live gauges (CDEC, USGS) run hourly for near-real-time flow;
#                    slower sources (groundwater, climate) run once daily
#                    because they only publish daily/periodically.
#   run_health_checks - Check database, disk, SSL, migrations, and sync freshness
#   prune_old_data    - Delete staging records and logs older than retention thresholds

OPENH2O_DIR=/opt/openh2o
OPENH2O_LOG_DIR=/opt/openh2o-logs

# Hourly: live stream / reservoir telemetry (near-real-time flow & stage)
0 * * * * $OPENH2O_DIR/scripts/run-sync.sh cdec usgs

# Daily 2:00 AM: slower periodic sources — groundwater (DWR WDL/SGMA) and climate (NOAA)
# (cimis/cnrfc/openet dropped 59-01: deactivated sources, no useful markers — see seed_data_sources.py)
0 2 * * * $OPENH2O_DIR/scripts/run-sync.sh dwr_wdl dwr_sgma noaa

# Health checks every 6 hours
0 */6 * * * cd $OPENH2O_DIR && docker compose exec -T web python manage.py run_health_checks >> $OPENH2O_LOG_DIR/health.log 2>&1

# Monthly data pruning on the 1st at 3:00 AM (--confirm required; default is dry-run)
0 3 1 * * cd $OPENH2O_DIR && docker compose exec -T web python manage.py prune_old_data --confirm >> $OPENH2O_LOG_DIR/prune.log 2>&1
